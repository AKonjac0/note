1.2 重置运算符

//1.this写法
struct point
{
    int elem;
    bool operator==(const point b) const
    {
        return this->elem == b.elem;
    }
    bool operator!=(const point b) const
    {
        return this->elem != b.elem;
    }
    bool operator<=(const point b) const
    {
        return this->elem <= b.elem;
    }
    bool operator<(const point b) const
    {
        return this->elem < b.elem;
    }
    bool operator>=(const point b) const
    {
        return this->elem >= b.elem;
    }
    bool operator>(const point b) const
    {
        return this->elem > b.elem;
    }
};
