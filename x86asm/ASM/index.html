    这段程序无法正确显示a[1],因为a[1]是动态变量，子程序结束后栈空间退回，动态变量死亡。 所以使用static. 
    #include<stdio.h>
int *f(int i){
    static int a[3]={10,20,30};
    return *a[i];
}
int main(){
    int *p;
    p=f(1);
    printf("%d\n",*p);
    return 0;
}

  10. gcc内嵌asm: 

    #include<stdio.h>
int main(){
    int input=1, output=0,temp=-1;
    printf("%d\n",output);
    printf("%d\n",temp);
    __asm__ __volatile__
    ("movl $0, %%eax; movl  %%eax, %1;     movl  %2, %%eax; movl  %%eax, %0; "
    :"=m"(output),"=m"(temp)    /* output */
    :"r"(input)     /* input */
    :"eax");  /* 描述符 */
    printf("%d\n",output);
    printf("%d\n",temp);
    return 0;
}
